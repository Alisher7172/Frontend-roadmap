Core Concepts

Variables

Containers for storing data values

// Variable declaration let message = "Hello World!"; const PI = 3.14159; var count = 10; // Using variables console.log(message); console.log("PI value:", PI);

Data Types

JavaScript has dynamic types

// Primitive types let name = "Alice"; // String let age = 30; // Number let isStudent = true; // Boolean let score = null; // Null let data; // Undefined // Complex types let colors = ["red", "green", "blue"]; // Array let person = { // Object firstName: "John", lastName: "Doe" };

Operators

Perform operations on variables and values

// Arithmetic let sum = 10 + 5; // 15 let product = 10 * 5; // 50 // Comparison let isEqual = 10 == "10"; // true (loose equality) let isStrictEqual = 10 === "10"; // false // Logical let andResult = true && false; // false let orResult = true || false; // true

Functions

Reusable blocks of code

// Function declaration function greet(name) { return `Hello, ${name}!`; } // Function expression const multiply = function(a, b) { return a * b; }; // Arrow function (ES6) const square = num => num * num; // Calling functions console.log(greet("Alice")); // "Hello, Alice!" console.log(multiply(4, 5)); // 20

Practical Examples

User Interaction

DOM Manipulation

// Change element content document.getElementById("demo").innerHTML = "New Content"; // Change element style document.querySelector(".box").style.backgroundColor = "blue"; // Handle button click document.getElementById("myBtn").addEventListener("click", function() { alert("Button clicked!"); });

Simple Calculator

function calculate() { let num1 = parseFloat(document.getElementById("num1").value); let num2 = parseFloat(document.getElementById("num2").value); let operator = document.getElementById("operator").value; let result; switch(operator) { case '+': result = num1 + num2; break; case '-': result = num1 - num2; break; case '*': result = num1 * num2; break; case '/': result = num1 / num2; break; default: result = "Invalid operator"; } document.getElementById("result").innerText = result; }
